template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

//Å‘åŒö–ñ”
//O(log max(a,b))
template<class T>
T GCD(T a, T b) {
	if (b == 0)return a;
	if (a < b)return GCD(b, a);
	return GCD(b, (T)(a % b));
}

long long A, B;
long long G;

int main() {
	cin >> A >> B;
	G = GCD<long long>(A, B);
	vector<long long> prime;
	for (long long p = 1; p * p <= G; p++) {
		if (G % p == 0) {
			prime.push_back(p);
			if (p * p != G)prime.push_back(G / p);
		}
	}
	sort(prime.begin(), prime.end());
	long long ans = 0;
	for (long long p : prime) {
		bool flag = false;
		for (long long n = 2; n * n <= p; n++) {
			if (p % n == 0) {
				flag = true;
				break;
			}
		}
		if (!flag)ans++;
	}
	cout << ans << endl;
}
