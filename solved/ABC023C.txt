const long long MOD = 1e9 + 7;
typedef long long ll;
typedef long double ld;
typedef pair<ll, ll> pll;
typedef pair<ld, ll> pdl;
typedef pair<ld, ld> pdd;
typedef vector<ll> VLL;
typedef vector<VLL> VVLL;
//typedef boost::multiprecision::cpp_int bigint;

ll H, W, K, N;
vector<pll> drops;
vector<pll> Y;
VLL order;
VVLL X;

int main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	cin >> H >> W >> K >> N;
	Y.resize(H + 1);
	for (ll y = 1; y <= H; y++) {
		Y[y].second = y;
		Y[y].first = 0;
	}
	drops.resize(N);
	for (ll n = 0; n < N; n++) {
		auto &d = drops[n];
		cin >> d.second >> d.first;
		Y[d.second].first++;
	}
	sort(Y.begin(), Y.end(), [](pll a, pll b) {
		return a.first < b.first;
	});
	order.resize(H + 1,0);
	for (ll y = 1; y <= H; y++) {
		order[Y[y].second] = y;
	}
	for (ll n = 0; n < N; n++) {
		drops[n].second = order[drops[n].second];
	}
	X.resize(W + 1);
	for (ll n = 0; n < N; n++) {
		X[drops[n].first].push_back(drops[n].second);
	}
	for (ll x = 1; x <= W; x++) {
		sort(X[x].begin(), X[x].end(), [](ll a, ll b) {
			return a < b;
		});
	}
	ll ans = 0;
	for (ll n = 0; n < N; n++) {
		ll x = drops[n].first, y = drops[n].second;
		if (X[x].size() + Y[y].first - 1 == K)ans++;
	}
	for (ll x = 1; x <= W; x++) {
		ll findy = K - X[x].size();
		if (findy < 0)continue;
		//bs<~<=be
		ll bs = 0, be = H;
		while (be - bs > 1) {
			ll m = (be + bs) / 2;
			if (Y[m].first < findy)bs = m;
			else be = m;
		}
		if (Y[be].first != findy)continue;
		//ts <= ~ < te
		ll ts = 1, te = H + 1;
		while (te - ts > 1) {
			ll m = (te + ts) / 2;
			if (Y[m].first <= findy)ts = m;
			else te = m;
		}
		//ƒAƒ‚ª‚ ‚éƒ}ƒX‚ðœ‚­
		ll plus = ts - be + 1;
		for (ll y : X[x]) {
			if (be <= y && y <= ts)plus--;
		}
		ans += plus;
	}
	cout << ans << "\n";
	return 0;
}
