using namespace std;
long long MOD = 1e9 + 7;
typedef pair<long long, long long> pii;

int N, Wmax;
vector<pii> items;   //d‚³A‰¿’l

int main() {
	cin >> N >> Wmax;
	items.resize(N);
	long long fmax = 0, smax = 0;
	for (int n = 0; n < N; n++) {
		cin >> items[n].second >> items[n].first;
		fmax = max(fmax, items[n].first);
		smax = max(smax, items[n].second);
	}
	if (N <= 30) {
		vector<pii> bag1, bag2;
		for (int code = 0; code < (1 << N / 2); code++) {
			long long w = 0, v = 0;
			for (int n = 0; n < N/2; n++) {
				if (code & (1 << n)) {
					w += items[n].first;
					v += items[n].second;
				}
			}
			if (w > Wmax)continue;
			bag1.push_back(pii(w, v));
		}
		for (int code = 0; code < (1 << (N - N / 2)); code++) {
			long long w = 0, v = 0;
			for (int n = 0; n < N - N / 2; n++) {
				if (code & (1 << n)) {
					w += items[n + N / 2].first;
					v += items[n + N / 2].second;
				}
			}
			if (w > Wmax)continue;
			bag2.push_back(pii(w, v));
		}
		long long ans = 0;
		sort(bag2.begin(), bag2.end(), [](pii a, pii b) {
			if (a.first != b.first)return a.first < b.first;
			return a.second < b.second;
		});
		for (int n = 1; n < bag2.size(); n++) {
			bag2[n].second = max(bag2[n - 1].second, bag2[n].second);
		}
		for (auto item1 : bag1) {
			long long x = Wmax - item1.first;
			int s = 0, e = bag2.size();
			while (e - s > 1) {
				int m = (e + s) / 2;
				if (bag2[m].first <= x)s = m;
				else e = m;
			}
			ans = max(ans, item1.second + bag2[s].second);
		}
		cout << ans << endl;
		return 0;
	}
	else if (fmax <= 1000) {
		vector<vector<long long>> DP;
		DP.resize(200, vector<long long>(200000+1, LLONG_MIN));
		DP[0][0] = 0;
		DP[0][items[0].first] = items[0].second;
		for (int n = 1; n < N; n++) {
			for (int w = 0; w <= 200000; w++) {
				DP[n][w] = DP[n - 1][w];
				if (w - items[n].first >= 0) {
					if(DP[n-1][w-items[n].first] != LLONG_MIN)DP[n][w] = max(DP[n][w], DP[n - 1][w - items[n].first] + items[n].second);
				}
			}
		}
		long long ans = 0;
		for (int w = 0; w <= min(Wmax,200000); w++) {
			ans = max(ans, DP[N-1][w]);
		}
		cout << ans << endl;
		return 0;
	}
	else {
		vector<vector<long long>> DP;
		DP.resize(200, vector<long long>(200001, INT_MAX));
		DP[0][0] = 0;
		DP[0][items[0].second] = items[0].first;
		for (int n = 1; n < N; n++) {
			for (int v = 0; v <= 200000; v++) {
				DP[n][v] = DP[n - 1][v];
				if (v - items[n].second >= 0) {
					DP[n][v] = min(DP[n][v], DP[n - 1][v - items[n].second] + items[n].first);
				}
			}
		}
		long long ans = 0;
		for (int v = 200000; v >= 0; v--) {
			if (DP[N - 1][v] <= Wmax) {
				ans = v;
				break;
			}
		}
		cout << ans << endl;
		return 0;
	}
}
