//O(log n)
template<class T>
T pow(T base, int p,T mod = -1) {
	if (p == 0)return 1;
	if (p == 1)return mod==-1?base%mod:base;
	int b = 1, c = 0;
	while (b <= p) {
		b *= 2;
		c++;
	}
	T* arr2 = new T[c];   //T[n] = T^(2^n)
	arr2[0] = mod == -1 ? base % mod : base;
	for (int n = 1; n < c; n++) {
		if (mod != -1)arr2[n] = (arr2[n - 1] * arr2[n - 1]) % mod;
		else arr2[n] = arr2[n - 1] * arr2[n - 1];
	}
	c--;
	T ans = arr2[c];
	b /= 2;
	p -= b;
	while (p > 0) {
		c--;
		b /= 2;
		if (p >= b) {
			if (mod != -1)ans = (ans * arr2[c]) % mod;
			else ans = ans * arr2[c];
			p -= b;
		}
	}
	return ans;
}

int N, M;
vector<int>* vec;
map<int, int> PMAP;
vector<vector<unsigned int>> dp;

void solve() {
	cin >> N >> M;
	if (abs(N) == 1) {
		cout << pow((long)2, M - 1) % MOD << endl;
		return;
	}
	int n = abs(N);
	int m = 1;
	int t = 2;
	while (n > 1 && t*t <= n) {
		if (n % t == 0) {
			if (PMAP.find(t) == PMAP.end())PMAP.insert(pair<int, int>(t, 1));
			else {
				PMAP[t]++;
				m = max(m, PMAP[t]);
			}
			n /= t;
		}
		else t++;
	}
	if (n != 1) {
		if (PMAP.find(n) == PMAP.end())PMAP.insert(pair<int, int>(n, 1));
		else {
			PMAP[n]++;
			m = max(m, PMAP[n]);
		}
	}
	dp.resize(m + 1);
	for (int i = 0; i <= m; i++) {
		dp[i].resize(M + 1);
		dp[i][1] = 1;
	}
	for (int n = 1; n <= M; n++)dp[1][n] = n;
	for (int n = 2; n <= m; n++) {
		for (int i = 2; i <= M; i++) {
			dp[n][i] = (dp[n-1][i]+dp[n][i-1]) % MOD;
		}
	}
	long long ans = 1;
	for (auto itr = PMAP.begin(); itr != PMAP.end(); itr++) {
		ans = (ans*dp[(*itr).second][M]) % MOD;
	}
	int p = pow((long long)2, M - 1,(long long)MOD);
	ans = (ans *  p) % MOD;
	cout << ans << endl;
}
