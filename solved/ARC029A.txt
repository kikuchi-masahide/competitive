#include <math.h>
#include <string>

using namespace std;

void solve();

int main() {
	solve();
	return 0;
}

int N;
int t[4];
void solve() {
	cin >> N;
	for (int i = 0; i < N; i++)cin >> t[i];
	int m = 200;
	for (int b = 0; b < pow(2, N); b++) {
		int t1 = 0, t2 = 0;
		for (int i = 0; i < N; i++) {   //0->1‚ÉA1->2‚É
			if ((b & (int)pow(2, i)) == 0)t1 += t[i];
			else t2 += t[i];
		}
		int mtemp = t1 < t2 ? t2 : t1;
		m = m < mtemp ? m : mtemp;
	}
	cout << m << endl;
	return;
}
