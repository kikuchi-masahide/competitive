using namespace std;

int N;
typedef pair<int, int> pii;
vector<pii> A;   //’†gA‡”Ô(0-indexed)
vector<int> AA;   //1`N‚Ì‘å‚«‚³‡‚ÅA‚ğ•\‚µ‚½‚à‚Ì

int main() {
	cin >> N;
	for (int n = 0; n < N; n++) {
		int a;
		cin >> a;
		A.push_back(pii(a, n));
	}
	sort(A.begin(), A.end(), [](pii a, pii b) {
		return a.first < b.first;
	});
	AA.resize(N);
	for (int n = 0; n < N; n++) {
		AA[A[n].second] = n;
	}
	long long ans = 0;
	for (int n = 1; n < N; n += 2) {
		if (AA[n] % 2 == 0)ans++;
	}
	cout << ans << endl;
	return 0;
}
