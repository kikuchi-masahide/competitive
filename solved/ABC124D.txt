using namespace std;

int N, K;
typedef pair<int, int> pii;
vector<pii> V;

int main() {
	string str;
	cin >> N >> K >> str;
	V.push_back(pii(str[0] - '0', 1));
	for (int n = 1; n < N; n++) {
		if (str[n] != str[n - 1])V.push_back(pii(str[n] - '0', 1));
		else V.back().second++;
	}
	if (V.back().first == 0)V.push_back(pii(1, 0));
	int counter = 0;
	int ans = 0;
	int start = 0, last = -1, k = 0;
	for (int n = 0; n < V.size(); n++) {
		if (V[n].first == 0 && k == K)break;
		counter += V[n].second;
		last++;
		if (V[n].first == 0)k++;
	}
	ans = counter;
	while (last < V.size()-1) {
		//ŽŸ‚Ì1‚Ü‚ÅL‚Î‚·
		counter += V[last+1].second + V[last+2].second;
		last += 2;
		if (V[start].first == 1) {
			counter -= V[start++].second;
			counter -= V[start++].second;
		}
		else counter -= V[start++].second;
		ans = max(ans, counter);
	}
	cout << ans << endl;
	return 0;
}
