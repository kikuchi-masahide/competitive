string S;

int main() {
	cin >> S >> K;
	N = S.size();
	vector<string> ss;   //K•¶š‚¾‚¯
	for (int n = 0; n < N; n++) {
		ss.push_back(S.substr(n, min(K, N - n)));
	}
	//2•¶š–ÚˆÈ~‚ğŒ©‚Ä¸‡
	sort(ss.begin(), ss.end(), [](string a, string b) {
		for (int n = 0; n < min(a.size(), b.size()); n++) {
			if (a[n] != b[n])return a[n] < b[n];
		}
		return a.size() < b.size();
		});
	set<string> sset;   //¡‚Ü‚ÅŒ©‚½•¶š—ñ‚Ì•Û‘¶—p
	for (int s = 0; s < ss.size(); s++) {
		for (int n = 1; n <= ss[s].size(); n++) {
			sset.insert(ss[s].substr(0, n));
			if (sset.size() == K) {
				cout << ss[s].substr(0, n) << endl;
				return 0;
			}
		}
	}
}
