#include <iostream>
#include <string>
#include <math.h>

using namespace std;

int main() {
	string S;
	cin >> S;
	long sum1 = 0,sum2 = 0;
	int l = S.length();
	int nums[10];
	for (int i = 0; i < l; i++)nums[i] = S[i] - '1' + 1;
	int max = pow(2, l - 1);
	for (int b = 0; b < max; b++) {
		long temp1 = nums[0];
		long temp2 = 0;
		for (int i = 0; i < l-1; i++) {
			int t = b & (int)pow(2,i);
			if (t != 0) {
				sum1 += temp1;
				sum2 += temp2;
				if (sum1 >= 100000000) {
					sum2 += floor(sum1 / 100000000);
					sum1 = sum1 % 100000000;
				}
				temp1 = nums[i + 1];
				temp2 = 0;
			}
			else {
				temp1 = 10 * temp1 + nums[i + 1];
				temp2 *= 10;
				if (temp1 >= 100000000) {
					temp2 += floor(temp1 / 100000000);
					temp1 = temp1 % 100000000;
				}
			}
		}
		sum1 += temp1;
		sum2 += temp2;
		if (sum1 >= 100000000) {
			sum2 += floor(sum1 / 100000000);
			sum1 = sum1 % 100000000;
		}
	}
	if (sum2 == 0) cout << sum1 << endl;
	else {
		int dig = floor(log10(sum1)) + 1;   //sum1‚ÌŒ…”
		cout << sum2;
		for (int i = 0; i < 8 - dig; i++)cout << "0";
		cout << sum1 << endl;
	}
}
