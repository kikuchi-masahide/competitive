//重み付きグラフ
//距離の部分をテンプレートに
template<class T>
class wGraph {
public:
	int N;
	vector<map<int, T> >* E;
	wGraph(int n) {
		N = n;
		E = new vector<map<int,T> >;
		E->resize(N);
	}
	~wGraph() {
		delete E;
	}
	void addEdge(int a, int b, T c) {
		E->at(a).insert(pair<int,T>(b, c));
	}
	bool isConnected(int a, int b) {
		return (E->at(a).find(b) != E->at(a).end());
	}
};
