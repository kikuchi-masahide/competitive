//O(log n)
template<class T>
T pow(T base,int p) {
	int b = 1, c = 0;
	while (b < p) {
		b *= 2;
		c++;
	}
	T* arr2 = new T[c];   //T[n] = T^(2^c)
	arr2[0] = base;
	for (int n = 1; n < c; n++)arr2[n] = arr2[n - 1] * arr2[n - 1];
	c--;
	T ans = arr2[c];
	b /= 2;
	p -= b;
	while (p > 0) {
		c--;
		b /= 2;
		if (p >= b) {
			ans *= arr2[c];
			p -= b;
		}
	}
	return ans;
}
