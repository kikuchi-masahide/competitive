using namespace std;
using namespace boost::multiprecision;
vector<int> vec;

//Å‘åŒö–ñ”
//O(log max(a,b))
template<class T>
T GCD(T a, T b) {
	if (b == 0)return a;
	return GCD(b, (T)(a%b));
}

void solve() {
	int N;
	cin >> N;
	vec.resize(N);
	for (int n = 0; n < N; n++) {
		int a, b;
		cin >> a >> b;
		vec[n] = GCD(a, b);
	}
	for (int i = 0; i < N; i++)cout << vec[i] << endl;
	return;
}
