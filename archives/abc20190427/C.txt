template<class T>
T GCD(T a, T b) {
	if (b == 0)return a;
	return GCD(b, (T)(a%b));
}

//Å‘åŒö–ñ”(•¡”)
//O(nlog max(a_i))?
template<class T>
T GCD(vector<T> v) {
	for (int n = 1; n < v.size(); n++) {
		v[n] = GCD(v[n], v[n - 1]);
	}
	return v[v.size() - 1];
}

int N;
vector<int> arr;
vector<int> garr2;
vector<int> garr1;

void solve() {
	cin >> N;
	if (N == 2) {
		int a, b; 
		cin >> a >> b;
		cout << max(a, b) << endl;
		return;
	}
	arr.resize(N);
	garr1.resize(N);
	garr2.resize(N);
	for (int n = 0; n < N; n++)cin >> arr[n];
	sort(arr.begin(),arr.end());
	garr1[0] = arr[0];
	for (int n = 1; n < N; n++) {
		garr1[n] = GCD(garr1[n-1],arr[n]);
	}
	garr2[N - 1] = arr[N - 1];
	for (int n = N - 2; n >= 0; n--) {
		garr2[n] = GCD(garr2[n+1],arr[n]);
	}
	int m = max(garr2[1],garr1[N-2]);
	for (int n = 1; n < N-1; n++) {
		m = max(m,GCD(garr1[n-1],garr2[n+1]));
	}
	cout << m << endl;
	return;
}
