#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int N;
string S;
int dp[200000][2];   //•:0A”’:1

void solve() {
	cin >> N >> S;
	if (S[0] == '.') dp[0][0] = 1, dp[0][1] = 0;
	else dp[0][0] = 0, dp[0][1] = 1;
	for (int n = 1; n < N; n++) {
		if (S[n] == '.') {
			dp[n][0] = min(dp[n-1][0],dp[n-1][1]) + 1;
			dp[n][1] = dp[n - 1][1];
		}
		else {
			dp[n][0] = min(dp[n - 1][0], dp[n - 1][1]);
			dp[n][1] = dp[n-1][1] + 1;
		}
	}
	cout << min(dp[N-1][0],dp[N-1][1]) << endl;
	return;
}
