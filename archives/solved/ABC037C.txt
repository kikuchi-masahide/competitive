#include <boost/multiprecision/cpp_int.hpp>

using namespace boost::multiprecision;

int N, K;
cpp_int A[100000];

void solve() {
	cin >> N >> K;
	string s;
	for (int n = 0; n < N; n++) {
		cin >> s;
		A[n] = cpp_int(s);
	}
	//—İÏ‚í
	for (int n = 1; n < N; n++)A[n] += A[n - 1];
	cpp_int ans = A[K-1];
	for (int n = K; n < N; n++)ans += A[n] - A[n - K];
	cout << ans << endl;
	return;
}
