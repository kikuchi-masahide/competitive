	solve();
	return 0;

using namespace std;
using namespace boost::multiprecision;

int N, M;
vector<int> P;
set<int> sums;
vector<int> ans;

void solve() {
	while (true) {
		cin >> N >> M;
		if (N == 0 && M == 0)break;
		P.clear();
		P.resize(N + 1);
		P[0] = 0;
		for (int i = 1; i <= N; i++)cin >> P[i];
		sums.clear();
		sort(P.begin(), P.end());
		for (int i = 0; i <= N; i++) {
			for (int j = i; j <= N; j++) {
				if (P[i] + P[j] <= M)sums.insert(P[i] + P[j]);
				else break;
			}
		}
		int i = 0;
		int mans = 0;
		for (auto itr = sums.begin(); itr != sums.end(); itr++) {
			int f = M - *itr;   //’T‚·‚×‚«”
			if (f < 0)continue;
			auto itr2 = sums.upper_bound(f);
			itr2--;
			mans = max(mans, *itr + *itr2);
		}
		ans.push_back(mans);
	}
	for (auto itr = ans.begin(); itr != ans.end(); itr++)cout << *itr << endl;
	return;
}
