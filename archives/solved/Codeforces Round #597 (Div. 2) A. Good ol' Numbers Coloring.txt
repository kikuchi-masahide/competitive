template<class T>
T GCD(T a, T b) {
	if (b == 0)return a;
	return GCD(b, (T)(a % b));
}
 
 
int main() {
	ll T;
	cin >> T;
	for (ll t = 0; t < T; t++) {
		ll a, b;
		cin >> a >> b;
		ll g = GCD(a, b);
		if (g == 1)cout << "Finite" << endl;
		else cout << "Infinite" << endl;
	}
	return 0;
}
