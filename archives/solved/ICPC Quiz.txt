#include <iostream>
#include <vector>
#include <queue>
#include <limits.h>
#include <algorithm>
#include <deque>

using namespace std;

void solve() {
	while (true) {
		int N, M;
		cin >> N >> M;
		if (N == 0 && M == 0)return;
		vector<int> vec;
		vector<int> vecmin;
		vec.resize(N, 0);
		vecmin.resize(N, 0);
		for (int m = 0; m < M; m++) {
			int S, K;
			cin >> S >> K;
			int c = 0;
			if (K == 1) {
				cin >> c;
				vec[--c] += S;
				vecmin[c] += S;
				continue;
			}
			for (int k = 0; k < K; k++) {
				cin >> c;
				vec[--c] += S;
			}
		}
		int ans = 0;
		for (int n = 0; n < N; n++) {   //nl–Ú‚ªmaxŠl“¾
			int mmin = INT_MAX;
			for (int m = 0; m < N; m++) {
				if (m == n)continue;
				mmin = min(mmin, vecmin[m]);
			}
			ans = max(ans, vec[n]-mmin);
		}
		cout << ans + 1 << endl;
	}
}
