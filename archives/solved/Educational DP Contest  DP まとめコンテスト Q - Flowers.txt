//‹óvector‚ğ—v‘f”N‚ÌsegTree‚Æ‚µ‚Äz‚Å‰Šú‰»
template<class T>
void segTree_init(vector<T>& vec, int N, T z) {
	int RN = 1;
	while (RN < N)RN *= 2;
	vec.resize(2 * RN, z);
}

//vec‚ğsegTree‚Æ‚µ‚Än”Ô–Ú‚ğx‚É
template<class T>
void segTree_set(vector<T>& vec, int n, T x, function<T(T, T)> f, int k = -1) {
	int RN = (vec.size() + 1) / 2;
	if (k == -1) {
		k = n + RN;
		vec[k] = x;
		segTree_set(vec, n, x, f, k / 2);
	}
	else if (k != 0) {
		vec[k] = f(vec[2 * k], vec[2 * k + 1]);
		segTree_set(vec, n, x, f, k / 2);
	}
}

//vec‚ªsegTree‚Ì‚ÌAn”Ô–Ú‚Ì—v‘f
template<class T>
T segTree_at(vector<T>& vec, int n) {
	int RN = (vec.size()) / 2;
	return vec[RN + n];
}

//get [a,b]
template<class T>
T segTree_get(vector<T>& vec, int a, int b, function<T(T, T)> f, T e, int k = 1, int l = 0, int r = -1) {
	if (r == -1)r = vec.size() / 2 - 1;
	if (b < l || r < a)return e;
	else if (a <= l && r <= b)return vec[k];
	else return f(
		segTree_get(vec, a, b, f, e, 2 * k, l, (l + r) / 2),
		segTree_get(vec, a, b, f, e, 2 * k + 1, (l + r) / 2 + 1, r)
	);
}

int N;
typedef pair<int, int> pii;
vector<int> A;
vector<long long> DP;
vector<pii> H;   //‚‚³A“Y‚¦š
std::function<long long(long long,long long)> lambda = [](long long a, long long b) {
	return max(a, b);
};

int main() {
	cin >> N;
	H.resize(N);
	A.resize(N);
	for (int n = 0; n < N; n++) {
		cin >> H[n].first;
		H[n].second = n;
	}
	for (int n = 0; n < N; n++)cin >> A[n];
	segTree_init(DP, N, (long long)0);
	sort(H.begin(), H.end(), [](pii a, pii b) {
		return a.first < b.first;
	});
	for (int n = 0; n < N; n++) {
		long long curmax = segTree_get(DP,0,H[n].second,lambda,(long long)0);
		segTree_set(DP, H[n].second, curmax+A[H[n].second], lambda);
	}
	cout << segTree_get(DP, 0, N - 1, lambda, (long long)0) << endl;
	return 0;
}
