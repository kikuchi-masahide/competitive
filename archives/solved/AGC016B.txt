using namespace std;

int N;
vector<int> A;

int main() {
	cin >> N;
	A.resize(N);
	for (int n = 0; n < N; n++)cin >> A[n];
	//A‚ÍˆêŽí—Þ‚©?
	sort(A.begin(), A.end());
	int kind = 1;
	for (int n = 1; n < N; n++) {
		if (A[n] != A[n - 1])kind++;
	}
	if (kind == 1) {
		int n = A[0];
		if (n + 1 == N)goto YES;
		else if (2 * n <= N)goto YES;
		else goto NO;
	}
	else if (kind == 2) {
		if (A[0] != A[N - 1] - 1)goto NO;
		//‚Ú‚Á‚¿‚Ì”‚ð”‚¦‚é
		int p = 0;
		for (int n = 0; n < N; n++) {
			if (A[n] == A[0])p++;
		}
		int q = N - p;
		int n = A[0];
		if (p >= n + 1)goto NO;
		if (q < 2 * (n + 1 - p))goto NO;
		goto YES;
	}
	else goto NO;
YES:
	cout << "Yes" << endl;
	return 0;
NO:
	cout << "No" << endl;
	return 0;
}
