template<class T>
void in(vector<T>& V) {
	for (auto itr = V.begin(); itr != V.end(); itr++)cin >> *itr;
	return;
}

template<class T>
void out(vector<T>& V) {
	for (T& itr : V)cout << itr << endl;
	return;
}

int N;
vector<string> A, B;
vector<vector<int>> X, Y;
vector<vector<int>> YY;
vector<set<int>> ZZ;
vector<set<int>> perm;   //uÔ‚¢v•Ó

int main() {
	cin >> N;
	A.resize(N);
	B.resize(N);
	in(A);
	in(B);
	X.resize(N, vector<int>());
	Y.resize(N, vector<int>());
	ZZ.resize(N, set<int>());
	YY.resize(N, vector<int>());
	perm.resize(N,set<int>());
	for (int x = 0; x < N; x++) {
		for (int y = 0; y < N; y++) {
			if (A[x][y] == '1') {
				X[x].push_back(y);
				YY[y].push_back(x);
			}
		}
	}
	for (int x = 0; x < N; x++) {
		for (int z = 0; z < N; z++) {
			if (B[x][z] == '0') {
				for (int y : X[x]) {
					perm[y].insert(z);
				}
			}
		}
	}
	for (int y = 0; y < N; y++) {
		for (int z = 0; z < N;z++) {
			if (perm[y].find(z) != perm[y].end())continue;
			Y[y].push_back(z);
			ZZ[z].insert(y);
		}
	}
	bool flag = true;
	for (int x = 0; x < N; x++) {
		for (int z = 0; z < N; z++) {
			if (B[x][z] == '1') {
				bool temp = false;
				for (int y : X[x]) {
					if (ZZ[z].find(y) != ZZ[z].end()) {
						temp = true;
						break;
					}
				}
				if (!temp) {
					flag = false;
					break;
				}
			}
		}
	}
	if (!flag)cout << -1 << endl;
	else {
		for (int y = 0; y < N; y++) {
			for (int z = 0; z < N; z++) {
				if (ZZ[z].find(y) != ZZ[z].end())cout << 1;
				else cout << 0;
			}
			cout << endl;
		}
	}
	return 0;
}
