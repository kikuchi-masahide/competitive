void solve() {
	cin  H  W;
	for (int y = 0; y  H; y++) {
		string s;
		cin  s;
		for (int x = 0; x  W; x++) {
			if (s[x] == 's') {
				start = pos(x, y);
				map[x][y] = true;
			}
			else if (s[x] == 'g') {
				goal = pos(x, y);
				map[x][y] = true;
			}
			else if (s[x] == '.')map[x][y] = true;

void bfs() {
	stackpos q;
	q.push(start);
	while (!q.empty()) {
		pos p = q.top();
		q.pop();
		if (rmap[p.x][p.y])continue;
		rmap[p.x][p.y] = true;
		pos vect[4] = {pos(1,0),pos(0,1),pos(-1,0),pos(0,-1)};
		for (int v = 0; v  4; v++) {
			pos n(p.x+vect[v].x,p.y+vect[v].y);
			if (!isIn(n))continue;
			if (!map[n.x][n.y])continue;
			q.push(n);
		}
	}
	return;
}
