void enumerateFactors(ll N, vector<ll>& V) {
	ll p = 2;
	while (N > 1) {
		if (p * p > N) {
			V.push_back(N);
			return;
		}
		if (N % p == 0) {
			V.push_back(p);
			N /= p;
		}
		else p++;
	}
	return;
}

int main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	ll N, K;
	cin >> N >> K;
	VLL P;
	enumerateFactors(N, P);
	if (P.size() < K) {
		cout << -1 << "\n";
		return 0;
	}
	ll cur = 1;
	for (ll k = 0; k < K - 1; k++) {
		cout << P[k] << " ";
		cur *= P[k];
	}
	cout << N / cur << "\n";
	return 0;
}
