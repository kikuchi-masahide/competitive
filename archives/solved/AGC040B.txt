ll N;
vector<pll> LR;
vector<ll> minleft;
vector<ll> minright;

ll calc(ll n) {
	if (n == 1) {
		ll ans = LR[1].second - LR[1].first + 1;
		if (LR[N].first <= minright[2])ans += minright[2] - LR[N].first + 1;
		return ans;
	}
	//ˆê”Ô¬‚³‚¢‚à‚Ì‚ðŽc‚·
	ll temp1 = min(LR[n].second, minleft[n - 1]);
	ll ans1 = max(minleft[n - 1] - LR[n].first + 1, (ll)0) + max(minright[n + 1] - LR[N].first + 1, (ll)0);
	//ˆê”Ô¬‚³‚¢‚à‚Ì‚ðˆÚ‚·
	ll temp2 = min(minright[n + 1], minleft[n - 1]);
	ll ans2 = max(LR[n].second - LR[n].first + 1, (ll)0) + max(temp2 - LR[N].first + 1, (ll)0);
	return max(ans1, ans2);
}

int main() {
	cin >> N;
	LR.resize(N + 1);
	for (ll n = 1; n <= N; n++) {
		cin >> LR[n].first >> LR[n].second;
	}
	sort(LR.begin(), LR.end(), [](pll a, pll b) {
		return a.first < b.first;
	});
	minleft.resize(N + 2);
	minleft[0] = LLONG_MAX;
	for (ll n = 1; n <= N; n++) {
		minleft[n] = min(minleft[n - 1], LR[n].second);
	}
	minleft[N + 1] = 0;
	minright.resize(N + 2);
	minright[N + 1] = LLONG_MAX;
	for (ll n = N; n >= 1; n--) {
		minright[n] = min(minright[n + 1], LR[n].second);
	}
	minright[0] = 0;
	ll ans = 0;
	for (ll n = 1; n <= N - 1; n++) {
		ll temp = calc(n);
		ans = max(ans, temp);
	}
	cout << ans << endl;
	return 0;
}
