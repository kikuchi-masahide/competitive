//‹óvector‚ğ—v‘f”N‚ÌsegTree‚Æ‚µ‚Äz‚Å‰Šú‰»
template<class T>
void segTree_init(vector<T>& vec, int N, T z) {
	int RN = 1;
	while (RN < N)RN *= 2;
	vec.resize(2 * RN, z);
}

//vec‚ğsegTree‚Æ‚µ‚Än”Ô–Ú‚ğx‚É
template<class T>
void segTree_set(vector<T>& vec, int n, T x, function<T(T, T)> f, int k = -1) {
	int RN = (vec.size() + 1) / 2;
	if (k == -1) {
		k = n + RN;
		vec[k] = x;
		segTree_set(vec, n, x, f, k / 2);
	}
	else if (k != 0) {
		vec[k] = f(vec[2 * k], vec[2 * k + 1]);
		segTree_set(vec, n, x, f, k / 2);
	}
}

//vec‚ªsegTree‚Ì‚ÌAn”Ô–Ú‚Ì—v‘f
template<class T>
T segTree_at(vector<T>& vec, int n) {
	int RN = (vec.size()) / 2;
	return vec[RN + n];
}

//get [a,b]
template<class T>
T segTree_get(vector<T>& vec, int a, int b, function<T(T, T)> f, T e, int k = 1, int l = 0, int r = -1) {
	if (r == -1)r = vec.size() / 2 - 1;
	if (b < l || r < a)return e;
	else if (a <= l && r <= b)return vec[k];
	else return f(
		segTree_get(vec, a, b, f, e, 2 * k, l, (l + r) / 2),
		segTree_get(vec, a, b, f, e, 2 * k + 1, (l + r) / 2 + 1, r)
	);
}

function<ll(ll, ll)> lambda = [](ll a, ll b) {
	return a + b;
};

int main() {
	std::ios::sync_with_stdio(false);
	std::cin.tie(0);
	ll N;
	vector<pll> H;
	cin >> N;
	H.resize(N+1);
	H[0] = pll(0, 0);
	for (ll n = 0; n < N; n++) {
		cin >> H[n+1].first;
		H[n+1].second = n+1;
	}
	sort(H.begin(), H.end(), [](pll a, pll b) {
		return a.first < b.first;
	});
	for (ll n = 0; n < N ; n++) {
		if (H[n].first == H[n + 1].first) {
			cout << -1 << "\n";
			return 0;
		}
	}
	VLL D(N + 1);   //“]“|”
	VLL seg;
	segTree_init(seg, N + 2, (ll)0);
	for (ll n = N; n >= 1; n--) {
		D[H[n].second] = segTree_get(seg, 0,H[n].second-1, lambda, (ll)0);
		segTree_set(seg, H[n].second, (ll)1, lambda);
	}
	ll ans = 0;
	for (ll n = 1; n <= N; n++) {
		ans += D[H[n].second] * H[n].first;
	}
	cout << ans << "\n";
	return 0;
}
