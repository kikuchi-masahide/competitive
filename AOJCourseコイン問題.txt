int N, M;
int C[20];
int dp[50001];   //dp[i] := i‚ð•¥‚¤Å¬–‡”

void solve() {
	cin >> N >> M;
	for (int i = 0; i < M; i++)cin >> C[i];
	dp[0] = 0;
	for (int i = 1; i <= N; i++) {
		dp[i] = 50000;
		for (int c = 0; c < M; c++) {
			if (i - C[c] < 0)continue;
			dp[i] = min(dp[i], dp[i - C[c]] + 1);
		}
	}
	cout << dp[N] << endl;
	return;
}
